matrix world;
matrix viewproj;

struct VS_INPUT
{
    vector position : POSITION;
    float3 normal   : NORMAL;
    vector diffuse  : COLOR;
    float2 texcoords: TEXCOORD;
};

struct VS_OUTPUT
{
    vector position : POSITION;
    vector diffuse  : COLOR;
    vector vpos     : TEXCOORD0;
    float3 normal   : TEXCOORD1;
    float2 texcoords: TEXCOORD2;
};

float3 light;

VS_OUTPUT main(VS_INPUT input)
{
    VS_OUTPUT output = (VS_OUTPUT)0;

    output.position = mul(input.position, mul(world, viewproj));

    float3 normal = mul(input.normal, (float3x3)world);
    normal = normalize(normal);

    float3 tolight = normalize(light);

    float s = dot(normal, tolight);
    if(s < 0.3f) s = 0.3f;

    output.diffuse = input.diffuse * s;

    output.vpos = mul(input.position, world);
    output.normal = mul(input.normal, (float3x3)world);
    output.texcoords = input.texcoords;

    return output;
}
